function iniciarClickAutomaticoSecundario() {
  if (intervaloClickSecundario) return;

  intervaloClickSecundario = setInterval(() => {
    game.pontosSecundario += game.valorClickSecundario;
    atualizaPontosSecundario();
  }, game.clickAutomaticoTempo);
}

function pararClickAutomaticoSecundario() {
  if (!intervaloClickSecundario) return;

  clearInterval(intervaloClickSecundario);
  intervaloClickSecundario = null;
}
